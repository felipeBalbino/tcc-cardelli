Com este script você poderá colocar em suas páginas um link que abre como se fosse um menu, exibindo um texto.

Clique aqui para ver este script em funcionamento.

Coloque o código abaixo entre as tags <HEAD> e </HEAD>:

<script language="JavaScript">
var MS=navigator.appVersion.indexOf("MSIE")
window.isIE4 =(MS>0) && ((parseInt(navigator.appVersion.substring(MS+5,MS+6)) >= 4) && (navigator.appVersion.indexOf("MSIE"))>0)
function checkExpand() {
if (""!=event.srcElement.id) {
var ch = event.srcElement.id + "Child"
var el = document.all[ch] 
if (null!=el) el.style.display = "none" == el.style.display ? "" : "none"
event.returnValue=false
}
}
</script>

Em seguida, coloque o código abaixo dentro da tag <BODY>:

<body onclick="checkExpand()" onunload="clearTimeout(window.tm)">

E finalmente, coloque o código abaixo no local onde o link vai aparecer:

<a CLASS="ex" href="#" title="Clique aqui" id="HG1">Crie seu Web site</a><div id="HG1Child"
style="display:none"><p>Web design, programação na Web, divulgação e muito mais.</font></p>
</div>
